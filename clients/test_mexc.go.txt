package main // WICHTIG: "package main"

import (
	"fmt"
	"log"

	ccxtpro "github.com/ccxt/ccxt/go/v4/pro"
)

// WICHTIG: Die "main"-Funktion als Startpunkt
func main() {
	exchangeName := "mexc"
	symbol := "BTC/USDT" 

	log.Printf("Erstelle Instanz für %s...", exchangeName)
	exchange := ccxtpro.CreateExchange(exchangeName, nil)
	exchange.SetVerbose(true)
	for {
		trades, err := exchange.WatchOHLCV(symbol)
		if err != nil {
			log.Printf("error in WatchTrades: %v", err)
			break 
		}
		
		fmt.Println("-----------------------------------------")
		fmt.Printf("Trade empfangen: %v\n", trades)
		fmt.Println("-----------------------------------------")
	}
}
