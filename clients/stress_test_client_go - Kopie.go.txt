type OrderBookLevel struct {
	Price  float64 `json:"price"`
	Amount float64 `json:"amount"`
}

type OrderBookUpdate struct {
	Exchange    string           `json:"exchange"`
	Symbol      string           `json:"symbol"`
	MarketType  string           `json:"market_type"`
	Timestamp   int64            `json:"timestamp"`
	GoTimestamp int64            `json:"go_timestamp"`
	Bids        []OrderBookLevel `json:"bids"`

	DataType    string           `json:"data_type"`
}

type TradeUpdate struct {
	Exchange    string  `json:"exchange"`
	Symbol      string  `json:"symbol"`
	MarketType  string  `json:"market_type"`
	Timestamp   int64   `json:"timestamp"`
	GoTimestamp int64   `json:"go_timestamp"`
	Price       float64 `json:"price"`
	Amount      float64 `json:"amount"`
	Side        string  `json:"side"`
	TradeID     string  `json:"trade_id"`
	DataType    string  `json:"data_type"`
}

func init() {
	// TRICK: Wir registrieren unsere lokalen Structs unter dem Namen des SERVERS.
	// So weiß GOB: "Aha, bybit-watcher...TradeUpdate ist eigentlich das hier."
	
	// 1. Die Basis-Structs mappen
	gob.RegisterName("bybit-watcher/internal/shared_types.TradeUpdate", TradeUpdate{})
	gob.RegisterName("bybit-watcher/internal/shared_types.OrderBookUpdate", OrderBookUpdate{})
	
	// 2. WICHTIG: Da der Server Slices (Listen) schickt, müssen wir GOB sagen, 
	// dass ein Slice dieses Typs in das Interface passt.
	// Wir registrieren hier einfach den lokalen Slice-Typ. Durch das Name-Mapping oben 
	// sollte GOB die Verbindung herstellen können.
	gob.Register([]*TradeUpdate{})
	gob.Register([]*OrderBookUpdate{})
}

// --- 3. CCXT LOGIK (KORRIGIERT: Pointer Dereferenzierung) ---
// ======================================================================

func getCCXTSymbols(handler string, marketType string, limit int) []string {
	exchangeID := strings.Split(handler, "_")[0]
	log.Printf("Lade Märkte für %s (%s) via CCXT...", strings.ToUpper(exchangeID), strings.ToUpper(marketType))

	var exc ccxt.IExchange
	options := map[string]interface{}{
		"defaultType": marketType,
	}
	config := map[string]interface{}{"options": options}

	switch exchangeID {
	case "binance":
		exc = ccxt.NewBinance(config)
	case "bybit":
		exc = ccxt.NewBybit(config)
	case "mexc":
		exc = ccxt.NewMexc(config)
	case "kucoin":
		exc = ccxt.NewKucoin(config)
	case "bitget":
		exc = ccxt.NewBitget(config)
	default:
		log.Printf("[WARN] Unbekannte Exchange ID für CCXT: %s", exchangeID)
		return []string{}
	}

	markets, err := exc.LoadMarkets()
	if err != nil {
		log.Printf("[ERROR] CCXT LoadMarkets fehlgeschlagen für %s: %v", exchangeID, err)
		return []string{}
	}

	var symbols []string

	for _, m := range markets {
		// Pointer Safety: Prüfen ob Active nil ist, bevor wir dereferenzieren
		if m.Active != nil && !*m.Active {
			continue
		}
		
		// WICHTIG: Symbol Pointer prüfen und dereferenzieren
		if m.Symbol == nil {
			continue
		}
		symbolStr := *m.Symbol // Hier holen wir den String-Wert aus dem Pointer

		if marketType == "spot" {
			// m.Spot ist auch ein Pointer (*bool)
			if m.Spot != nil && *m.Spot && strings.Contains(symbolStr, "/USDT") {
				symbols = append(symbols, symbolStr)
			}
		}

		if marketType == "swap" {
			// m.Swap und m.Settle sind Pointer
			if m.Swap != nil && *m.Swap && m.Settle != nil && *m.Settle == "USDT" {
				symbols = append(symbols, symbolStr)
			}
		}
	}

	if len(symbols) == 0 {
		log.Printf("[WARN] Keine passenden Märkte gefunden für %s %s", exchangeID, marketType)
		return []string{}
	}

	rand.Seed(time.Now().UnixNano())
	rand.Shuffle(len(symbols), func(i, j int) { symbols[i], symbols[j] = symbols[j], symbols[i] })

	if len(symbols) > limit {
		return symbols[:limit]
	}
	return symbols
}

type Statistics struct {
	mu             sync.Mutex
	TradeCount     int
	TotalTrades    int
	TradeLatencies []int64

	ObCount     int
	TotalOb     int
	ObLatencies []int64
}

func (s *Statistics) AddTrades(count int, lats []int64) {
	s.mu.Lock()
	defer s.mu.Unlock()
	s.TradeCount += count
	s.TotalTrades += count
	s.TradeLatencies = append(s.TradeLatencies, lats...)
}

func (s *Statistics) AddOrderBooks(count int, lats []int64) {
	s.mu.Lock()
	defer s.mu.Unlock()
	s.ObCount += count
	s.TotalOb += count
	s.ObLatencies = append(s.ObLatencies, lats...)
}

func (s *Statistics) ResetAndPrint() {
	s.mu.Lock()
	defer s.mu.Unlock()

	fmt.Println("\n==================================================")
	printStatGroup("Trades", s.TradeCount, s.TotalTrades, s.TradeLatencies)
	printStatGroup("OrderBooks", s.ObCount, s.TotalOb, s.ObLatencies)
	fmt.Println("==================================================")

	s.TradeCount = 0
	s.TradeLatencies = s.TradeLatencies[:0]
	s.ObCount = 0
	s.ObLatencies = s.ObLatencies[:0]
}

func printStatGroup(label string, current, total int, lats []int64) {
	if total == 0 && current == 0 {
		return
	}
	rate := float64(current) / 10.0
	fmt.Printf("[STATS|%-10s] Rate: %d in 10s (%.2f/s) | Gesamt: %d\n", label, current, rate, total)

	if len(lats) > 0 {
		sort.Slice(lats, func(i, j int) bool { return lats[i] < lats[j] })
		var sum int64
		for _, l := range lats {
			sum += l
		}
		avg := float64(sum) / float64(len(lats))
		max := float64(lats[len(lats)-1])
		p95Index := int(math.Floor(float64(len(lats)) * 0.95))
		p95 := float64(lats[p95Index])

		fmt.Printf("                   └─ Latenz (ms): avg=%.2f, p95=%.2f, max=%.0f\n", avg, p95, max)
	}
}

type BrokerClient struct {
	socket  zmq4.Socket
	stats   *Statistics
	stopCh  chan struct{}
	zmqAddr string
}

func NewBrokerClient(addr string) *BrokerClient {
	return &BrokerClient{
		zmqAddr: addr,
		stats:   &Statistics{},
		stopCh:  make(chan struct{}),
	}
}

func (bc *BrokerClient) Connect() {
	clientID := fmt.Sprintf("go-stress-%d", time.Now().UnixNano())
	bc.socket = zmq4.NewDealer(context.Background(), zmq4.WithID(zmq4.SocketIdentity(clientID)))
	if err := bc.socket.Dial(bc.zmqAddr); err != nil {
		log.Fatalf("ZMQ Connect Fehler: %v", err)
	}
	log.Printf("Client mit ID '%s' verbunden mit %s", clientID, bc.zmqAddr)
}

func (bc *BrokerClient) Start() {
	go bc.listenLoop()
	go bc.statsLoop()
}

func (bc *BrokerClient) listenLoop() {
	for {
		select {
		case <-bc.stopCh:
			return
		default:
			msg, err := bc.socket.Recv()
			if err != nil {
				log.Printf("ZMQ Recv Error: %v", err)
				continue
			}

			// Dealer empfängt vom Router Frames.
			// Struktur High-Speed: [Header("T"/"O"), Payload(GOB)]
			// Struktur Ping (Legacy): [EmptyByte, Payload(JSON)]
			
			if len(msg.Frames) < 2 {
				continue
			}

			headerFrame := msg.Frames[len(msg.Frames)-2]
			payloadFrame := msg.Frames[len(msg.Frames)-1]
			
			msgType := string(headerFrame)
			recvTs := time.Now().UnixMilli()

			// --- KORREKTUR: PING HANDLING ---
			// Wenn der Header leer ist, ist es wahrscheinlich eine Admin-Nachricht (Ping)
			if len(headerFrame) == 0 {
				if len(payloadFrame) > 0 && payloadFrame[0] == '{' {
					// JSON Check
					if bytes.Contains(payloadFrame, []byte("ping")) {
						bc.sendPong()
					}
				}
				continue
			}

			// --- DATEN HANDLING ---
			switch msgType {
			case "T": // Trades
				var trades []*TradeUpdate
				dec := gob.NewDecoder(bytes.NewReader(payloadFrame))
				if err := dec.Decode(&trades); err != nil {
					log.Printf("[ERR] Trade Decode Failed: %v", err)
					continue
				}
				
				lats := make([]int64, 0, len(trades))
				for _, t := range trades {
					if t.GoTimestamp > 0 {
						lats = append(lats, recvTs-t.GoTimestamp)
					}
				}
				bc.stats.AddTrades(len(trades), lats)

			case "O": // OrderBooks
				var obs []*OrderBookUpdate
				dec := gob.NewDecoder(bytes.NewReader(payloadFrame))
				if err := dec.Decode(&obs); err != nil {
					log.Printf("[ERR] OB Decode Failed: %v", err)
					continue
				}
				
				lats := make([]int64, 0, len(obs))
				for _, ob := range obs {
					if ob.GoTimestamp > 0 {
						lats = append(lats, recvTs-ob.GoTimestamp)
					}
				}
				bc.stats.AddOrderBooks(len(obs), lats)

			default:
				// log.Printf("Unbekannter Header: %s", msgType)
			}
		}
	}
}

func (bc *BrokerClient) statsLoop() {
	ticker := time.NewTicker(10 * time.Second)
	defer ticker.Stop()
	for {
		select {
		case <-bc.stopCh:
			return
		case <-ticker.C:
			bc.stats.ResetAndPrint()
		}
	}
}

func (bc *BrokerClient) SubscribeBulk(handler string, symbols []string, marketType, dataType string, depth int) {
	log.Printf("Sende Bulk-Subscribe an '%s' für %d Symbole (Typ: %s, Tiefe: %d)...", handler, len(symbols), dataType, depth)
	
	req := map[string]interface{}{
		"action":      "subscribe_bulk",
		"exchange":    handler,
		"symbols":     symbols,
		"market_type": marketType,
		"data_type":   dataType,
		"encoding":    "gob",
	}
	
	if dataType == "orderbooks" && depth > 0 {
		req["depth"] = depth
	}

	reqBytes, _ := json.Marshal(req)
	bc.socket.Send(zmq4.NewMsg(reqBytes))
}

func (bc *BrokerClient) sendPong() {
	pong, _ := json.Marshal(map[string]string{"message": "pong"})
	bc.socket.Send(zmq4.NewMsg(pong))
}

func (bc *BrokerClient) Stop() {
	close(bc.stopCh)
	bc.socket.Close()
}

func main() {
	log.Println("Starte Go Stresstest-Client (CCXT Version)...")
	
	client := NewBrokerClient("tcp://localhost:5555")
	client.Connect()
	client.Start()

	totalSubs := 0

	for _, job := range StressTestConfig {
		log.Printf("\n--- Verarbeite Job: %s | %s | %s ---", job.Handler, job.MarketType, job.DataType)

		symbols := getCCXTSymbols(job.Handler, job.MarketType, job.SymbolLimit)
		
		if len(symbols) > 0 {
			client.SubscribeBulk(job.Handler, symbols, job.MarketType, job.DataType, job.Depth)
			totalSubs += len(symbols)
			time.Sleep(500 * time.Millisecond)
		} else {
			log.Println("Keine Symbole gefunden.")
		}
	}

	log.Printf("\nAlle Jobs verarbeitet. %d Gesamt-Abonnements gesendet.", totalSubs)
	log.Println("Lausche auf Daten... Drücke Ctrl+C zum Beenden.")

	select {}
}
